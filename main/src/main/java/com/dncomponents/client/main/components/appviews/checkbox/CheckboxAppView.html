Add to CheckBoxSelectionGroup as checkboxes

Java 
            
CheckBoxSelectionGroup<Fruit> checkBoxGroup = new CheckBoxSelectionGroup<>();
checkBoxGroup.addItems(Fruit.getFruits(4).stream().map(CheckBox::new).collect(Collectors.toList()));
checkBoxGroup.getItems().forEach(e -> javaPanel.appendChild(e.asElement()));
Typescript 
             
let checkBoxGroup: CheckBoxSelectionGroup<Fruit> = new CheckBoxSelectionGroup<any>();
checkBoxGroup.addItems(Fruit.getFruits(4).stream().map((e) => new CheckBox().setUserObject(e)).collect(Collectors.toList()));
checkBoxGroup.getItems().forEach((e) => this.javaPanel.appendChild(e.asElement()));

Events

Logs selection events.

Java 
            
CheckBoxSelectionGroup<Fruit> eventsCheckboxGroup = new CheckBoxSelectionGroup<>();
eventsCheckboxGroup.addEntityItems(Fruit.getFruits(4));
eventsCheckboxGroup.getItems().forEach(e -> eventsPanel.appendChild(e.asElement()));

eventsCheckboxGroup.addSelectionHandler(event -> {
    eventsTa.setValue("");
    for (CheckBox<Fruit> checkBox : event.getSelectedItem()) {
        eventsTa.append(checkBox.getUserObject().getName() + " - " + checkBox.getUserObject().getDescription() + "\n");
    }
});
Typescript 
             
let eventsCheckboxGroup: CheckBoxSelectionGroup<Fruit> = new CheckBoxSelectionGroup<any>();
eventsCheckboxGroup.addEntityItems(Fruit.getFruits(4));
eventsCheckboxGroup.getItems().forEach(e => this.eventsPanel.appendChild(e.asElement()));

eventsCheckboxGroup.addSelectionHandler(SelectionHandler.onSelection(evt => {
    this.eventsTa.setValue('');
    evt.selection.forEach(f => {
        let checkBox: CheckBox<Fruit> = f;
        this.eventsTa.append(checkBox.getUserObject().name + ' - ' + checkBox.getUserObject().description + '\n');
    });
}));
