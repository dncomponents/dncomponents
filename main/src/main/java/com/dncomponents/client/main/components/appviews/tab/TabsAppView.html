Renderer

Create custom renderer for complex objects.

Java 
            
tabRenderer.setTabItemRenderer(new TabItem.RenderTabItem<Fruit>() {
@Override
public void render(Fruit userObject, TabItemViewSlots slots) {
        slots.getTitle().textContent = userObject.getName();
        slots.getContent().innerHTML = "<h2>" + SafeHtmlUtils.htmlEscape(userObject.getDescription()) + "</h2>";
    }
});
for (Fruit fruit : Fruit.fruits)
    tabRenderer.addItem(new TabItem<>(tabRenderer, fruit));
Typescript 
            
this.tabRenderer.setTabItemRenderer({
    render(userObject: Fruit, slots: TabItemViewSlots): void {
        slots.getTitle().textContent = userObject.name;
        slots.getContent().innerHTML = '<h4>' + userObject.description + '</h4>';
    }
});

Fruit.getFruits(5).forEach(fruit =>
        this.tabRenderer.addItem(new TabItem<Fruit>(this.tabRenderer).setUserObject(fruit)));

Java 
            
@UiField
Tab tabEvents;
@UiField
TextArea ta;

tabEvents.addSelectionHandler((SelectionHandler<TabItem<Object>>) event ->
        ta.append("SelectionEvent - tab: " + event.getSelectedItem().getOrder() + "\n"));

tabEvents.addBeforeSelectionHandler((BeforeSelectionHandler<TabItem<Object>>) event ->
        ta.append("BeforeSelectionEvent - tab: " + event.getItem().getOrder() + "\n"));
Typescript

    this.ta.append('SelectionEvent - tab: ' + evt.selection.getOrder() + '\n');
}));
this.tabEvents.addBeforeSelectionHandler(BeforeSelectionHandler.onBeforeSelection(evt => {
    this.ta.append('BeforeSelectionEvent - tab: ' + evt.selection.getOrder() + '\n');
