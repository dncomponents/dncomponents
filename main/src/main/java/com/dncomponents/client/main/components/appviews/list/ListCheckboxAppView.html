@UiField
public ListData<Fruit, String> list;

list.getRowCellConfig().setFieldGetter(fruit -> fruit.getName());
list.getRowCellConfig().setCellFactory(c -> new ListCellCheckbox<>());
list.setRowsData(Fruit.getFruits(2000));
list.drawData();
