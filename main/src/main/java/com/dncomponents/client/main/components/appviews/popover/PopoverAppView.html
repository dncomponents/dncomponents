Renderer
Renderer
Java 
            
@UiField
Button<any> rendererBtn;

Popover.PopoverRenderer<Fruit> renderer = (fruit, slots) -> {
    slots.getTitle().innerHTML = "<b>" + fruit.getName() + "</b>";
    slots.getContent().innerHTML = "<p>" + fruit.getDescription() + "</p>";
};

Popover<Fruit> popover = new Popover<>(rendererBtn.asElement(), BaseTooltip.Orientation.RIGHT);
popover.setRenderer(renderer);
popover.setUserObject(new Fruit("Apple", "Sweet fruit!"));
Typescript 
            
@UiField
rendererBtn: Button<any>;

renderer: PopoverRenderer<Fruit> = {
    render(fruit: Fruit, slots: PopoverViewSlots): void {
        slots.getTitle().innerHTML = '<b>' + fruit.name + '</b>';
        slots.getMainSlot().innerHTML = '<p>' + fruit.description + '</p>';
    }
};

let popover: Popover<Fruit> = new Popover<Fruit>()
    .setOwner(this.rendererBtn.asElement())
    .setOrientation(Orientation.RIGHT);
popover.setRenderer(this.renderer);
popover.setUserObject(new Fruit('Apple', 'Sweet fruit!'));
