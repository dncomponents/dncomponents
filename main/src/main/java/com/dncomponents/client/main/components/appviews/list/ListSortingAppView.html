Java 
            
@UiField
ListData<Fruit, String> list;
@UiField
Button<Boolean> sortAscBtn;
@UiField
Button<Boolean> sortDescBtn;
@UiField
Button<Boolean> sortOrgBtn;
@UiField
CheckBox<String> sortableChb;

list.getRowCellConfig().setFieldGetter(Fruit::getName);
list.getRowCellConfig().setFieldSetter(Fruit::setName);
list.getRowCellConfig().setCellRenderer(r ->
        r.valuePanel.innerHTML = r.cell.getModel().getName() +
                " <b>" + r.cell.getModel().getDescription() + "</b>");
list.setRowsData(Fruit.getFruits(2000));
list.drawData();
sortableChb.setValue(list.isSortable(), false);
sortableChb.addValueChangeHandler(event -> list.setSortable(event.getValue()));
sortAscBtn.setUserObject(Boolean.TRUE);
sortDescBtn.setUserObject(Boolean.FALSE);
sortOrgBtn.setUserObject(null);
ClickHandler sortHandler = e -> list.sort((Boolean) BaseComponent.source(e).getUserObject());
sortAscBtn.addClickHandler(sortHandler);
sortDescBtn.addClickHandler(sortHandler);
sortOrgBtn.addClickHandler(sortHandler);
Typescript 
             
@UiField
list: ListData<Fruit, string>;
@UiField
sortAscBtn: Button<boolean>;
@UiField
sortDescBtn: Button<boolean>;
@UiField
sortOrgBtn: Button<boolean>;
@UiField
sortableChb: CheckBox<string>;

this.list.getRowCellConfig().setFieldGetter((p) => {
    return p.name;
});
this.list.getRowCellConfig().setFieldSetter((p, name) => {
    return p.name = (name);
});
this.list.getRowCellConfig()

    .setCellRenderer({setValue: (r) => r.valuePanel.innerHTML = r.cell.getModel().name + ' <b>' + r.cell.getModel().name + '</b>'});
this.list.setRowsData(Fruit.getFruits(2000));
this.list.drawData();
this.sortableChb.setValue(this.list.isSortable(), false);
this.sortableChb.addValueChangeHandler(ValueChangeHandler.onValueChange(evt => {
    this.list.setSortable(evt.value);
}));
this.sortAscBtn.setUserObject(javaemul.internal.BooleanHelper.TRUE);
this.sortDescBtn.setUserObject(javaemul.internal.BooleanHelper.FALSE);
this.sortOrgBtn.setUserObject(null);
let sortHandler: ClickHandler = ClickHandler.onClick(evt => {
    this.list.sort(BaseComponent.source(evt).getUserObject());
});
this.sortAscBtn.addClickHandler(sortHandler);
this.sortDescBtn.addClickHandler(sortHandler);
this.sortOrgBtn.addClickHandler(sortHandler);
