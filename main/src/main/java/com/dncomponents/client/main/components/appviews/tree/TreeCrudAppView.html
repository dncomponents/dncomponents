Typescript 
            
let personCellConfig: TreeCellConfig<Person, string> = new TreeCellConfig<any, any>((o) => o.getUserObject().getName());
personCellConfig.setFieldSetter((n, s) => n.getUserObject().setName(s));
personCellConfig.setCellEditor(new DefaultCellEditor<any, any>(new TextArea()));
personCellConfig.setCellRenderer({setValue: (r) => r.valuePanel.innerHTML = '' + r.value + ' Person'});

let fruitCellConfig: TreeCellConfig<Fruit, string> = <any>(new TreeCellConfig<any, any>((n) => n.getUserObject().getName()));
fruitCellConfig.setFieldSetter((node, s) => node.getUserObject().setName(s));
fruitCellConfig.setClazz('string');
fruitCellConfig.setCellRenderer({
    setValue: (r) => {
        r.valuePanel.innerHTML = '' + r.value;
        r.valuePanel.style.color = 'blue';
    }
});
